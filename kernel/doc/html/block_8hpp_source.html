637  typedef typename ParamSpec< callable_t >::ptr callable_ptr;
638  // Вызывать блоки и подпрограммы в зависимости от приоритета
639  auto c_obj = Scheduler::template get_nextcall_objs
640  < callable_ptr, map_t, subprogram_map >
641  ( _children, _subprog, time );
642  for( auto & c: c_obj ) {
643  // Получаем новые значения переменных
644  parameter_map vm;
645  try {
646  auto b_p = std::dynamic_pointer_cast< block_t >( c );
647  if( !b_p ) {
648  // Это подпрограмма
649  if( c->is_powered( _vars ) )
650  { vm = (*c)( time, _vars, vars ); }
651 
652  } else {
653  // Это блп
654  bool pow = b_p->is_powered( _vars );
655  if( pow && b_p->get_prev_power() ) {
656  // Постоянная работа
657  vm = (*b_p)( time, _vars, vars );
658  } else {
659  if( pow && !b_p->get_prev_power() ) {
660  // Передний фронт питания
661  vm = b_p->write_poweron_vars();
662  // Сбрасываем время ожидания блока
663  b_p->touch();
664  } else if( b_p->get_prev_power() ) {
666  vm = b_p->power_off_children();
667  }
668  }
669  b_p->set_prev_power( pow );
670  }
671 
672  } catch( typename FormulaSolver::invalid_args & e ) {
673  _errors.push_back(
674  "Недостаточное количество операндов в выражении \""
675  + c->get_power_cond() + "\" для оператора \""
676  + e.op + "\" в позиции " + std::to_string( e.position ) );
677  _err_blocks.insert( c->get_name() );
678  vm.clear();
679  } catch( typename FormulaSolver::invalid_operation & e ) {
680  _errors.push_back(
681  "Неверная операция \"" + e.operation + "\" в выражении \""
682  + c->get_power_cond() + "\"" );
683  _err_blocks.insert( c->get_name() );
684  vm.clear();
685  } catch( exception::invalid_var_name & e ) {
686  _errors.push_back(
687  std::string( "Ошибка при выполнении модуля " )
688  + c->get_name() + ". Произошло обращение к "
689  + "отсутствующей переменной \"" + e.get_name() + "\"" );
690  _err_blocks.insert( c->get_name() );
691  vm.clear();
692  } catch( std::out_of_range & e ) {
693  // Скорее всего внутри подпрограммы обратились к
694  // отсутствующей переменной
695  _errors.push_back(
696  std::string( "Ошибка при выполнении модуля " )
697  + c->get_name() + ". Скорее всего произошло"
698  + " обращение к отсутствующей переменной" );
699  _err_blocks.insert( c->get_name() );
700  vm.clear();
701  } catch( exception::invalid_type & e ) {
702  _errors.push_back(
703  std::string( "Ошибка при выполнении модуля " )
704  + c->get_name() + ". Произошло неверное приведение "
705  + "типа для переменной \"" + e.get_name() + "\"" );
706  _err_blocks.insert( c->get_name() );
707  vm.clear();
708  } catch( std::bad_cast & e ) {

710  // преобразование типа
711  _errors.push_back(
712  std::string( "Ошибка при выполнении модуля " )
713  + c->get_name() + ". Скорее всего произошло"

715  _err_blocks.insert( c->get_name() );
